    use std::sync::Mutex;
    use chrono::{DateTime, Utc, Duration};

    /// 速率池类型
    #[derive(Debug, Clone, Copy, PartialEq, Eq)]
    pub enum RatePool { Core, Search }

    /// 单池状态
    struct PoolState {
        remaining: u32,
        resets_at: DateTime<Utc>,
    }

    /// 速率预算管理器 — core(5000/h) + search(30/min)
    pub struct RateBudget {
        core: Mutex<PoolState>,
        search: Mutex<PoolState>,
    }

    impl RateBudget {
        pub fn new() -> Self {
            Self {
                core: Mutex::new(PoolState { remaining: 5000, resets_at: Utc::now() + Duration::hours(1) }),
                search: Mutex::new(PoolState { remaining: 30, resets_at: Utc::now() + Duration::minutes(1) }),
            }
        }

        /// 检查是否可发起请求，返回错误信息若超限
        pub fn check(&self, pool: RatePool) -> Result<(), RateError> { ... }

        /// 记录一次请求消耗
        pub fn record(&self, pool: RatePool) { ... }

        /// 从 GitHub 响应头更新剩余配额
        pub fn update_from_headers(&self, pool: RatePool, remaining: u32, resets_at: DateTime<Utc>) { ... }
    }
    ```

    #[derive(Debug, thiserror::Error)]
    pub enum RateError {
        #[error("Core API 速率限制，重置于 {0}")]
        CoreExceeded(DateTime<Utc>),
        #[error("Search API 速率限制，重置于 {0}")]
        SearchExceeded(DateTime<Utc>),
    }
    ```

       pub struct SearchQuery {
           pub language: Option<String>,
           pub topic: Option<String>,
           pub min_stars: Option<u32>,
           pub sort: SearchSort,        // stars | updated | created
           pub order: SortOrder,        // asc | desc

       }

    4. `SearchResult` struct:
       ```rust
       pub struct SearchResult {
           pub items: Vec<domain::repository::Repository>,
           pub total_count: u64,
           pub incomplete_results: bool,
       }
       ```

    5. `pub async fn search_repositories(token: &str, query: &SearchQuery, budget: &RateBudget) -> Result<SearchResult, SearchError>`:
       - 调用 budget.check(RatePool::Search)
       - 构建搜索 query string：
         - 若有 language: 加 "language:{lang}"
         - 若有 topic: 加 "topic:{topic}"
         - 若有 min_stars: 加 "stars:>={min}"
         - 默认加 "archived:false"
       - 创建 octocrab client（复用 personal_token 模式）
       - 调用 octocrab.search().repos(&query_str)
         - .sort(query.sort.as_str())
         - .order(query.order.as_str())
         - .per_page(query.per_page)
         - .page(query.page)
       - budget.record(RatePool::Search)
       - 将 octocrab::models::Repository 映射为 domain::repository::Repository
       - 返回 SearchResult

    6. `SearchError` enum:
       ```rust
       pub enum SearchError {
           RateLimited(String),
           NetworkError(String),
           ApiError(String),
       }
       ```

    7. octocrab::models::Repository → domain::repository::Repository 映射：
       - repo_id: r.id.map(|id| id.0 as i64).unwrap_or(0)
       - full_name: r.full_name.unwrap_or_default()
