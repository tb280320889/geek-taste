    pub enum ResourceKind {
        McpServer,
        Skill,
        AgentFramework,
        Other(String),

    impl Display → "MCP_SERVER", "SKILL", "AGENT_FRAMEWORK", other as-is

    pub enum CurationLevel {
        UserCurated,   // 用户手动添加
        SystemDiscovered, // GitHub Search 发现
    }
    impl Display → "USER_CURATED", "SYSTEM_DISCOVERED"
    impl from_str(s: &str) → Option<Self>

    // ResourceTag — 资源标签
    pub struct ResourceTag {
        pub resource_id: String,
        pub tag_type: String,   // "language" | "framework" | "agent" | "free"
        pub tag_value: String,
    }

    // Resource — 对应 resources 表
    pub struct Resource {
        pub resource_id: String,
        pub source_repo_id: Option<i64>,  // FK → repositories, 可为 null
        pub resource_kind: ResourceKind,
        pub title: String,
        pub summary: Option<String>,
        pub source_url: String,
        pub languages: Vec<String>,       // languages_json 反序列化
        pub framework_tags: Vec<String>,  // framework_tags_json 反序列化
        pub agent_tags: Vec<String>,      // agent_tags_json 反序列化
        pub curation_level: CurationLevel,
        pub last_scored_at: Option<String>,
        pub is_active: bool,
    }
    // Resource 包含 fn signal_key(&self) -> String (用于 RESOURCE_EMERGED 信号去重)

    // ResourceScore — 评分结果
    pub struct ResourceScore {
        pub total: f64,
        pub stack_relevance: f64,
        pub star_delta_norm: f64,
        pub recency_norm: f64,
    }

    // ResourceScore 包含:
    pub fn compute_resource_score(
        stack_relevance: f64,   // 0.0-1.0
        star_delta_norm: f64,   // 0.0-1.0
        recency_norm: f64,      // 0.0-1.0
    ) -> ResourceScore
    // 公式: 0.4 * stack_relevance + 0.35 * star_delta_norm + 0.25 * recency_norm

    // 推荐解释枚举
    pub enum RecommendationReason {
        LanguageMatch(String),     // "你关注 Rust，这是 Rust 生态的 MCP server"

        UserCurated,               // "你已将此资源加入精选列表"
    }
    impl RecommendationReason → fn to_template(&self, resource_title: &str) -> String
    // 返回中文模板字符串

    // V003: Phase 4 资源表
    M::up(r#"
        CREATE TABLE IF NOT EXISTS resources (
            resource_id TEXT PRIMARY KEY,
            source_repo_id INTEGER REFERENCES repositories(repo_id),
            resource_kind TEXT NOT NULL,
            title TEXT NOT NULL,
            summary TEXT,
            source_url TEXT NOT NULL,
            languages_json TEXT NOT NULL DEFAULT '[]',
            framework_tags_json TEXT NOT NULL DEFAULT '[]',
            agent_tags_json TEXT NOT NULL DEFAULT '[]',
            curation_level TEXT NOT NULL DEFAULT 'SYSTEM_DISCOVERED',
            last_scored_at TEXT,
            is_active INTEGER NOT NULL DEFAULT 1
        );

        CREATE TABLE IF NOT EXISTS resource_tags (
            resource_id TEXT NOT NULL REFERENCES resources(resource_id),
            tag_type TEXT NOT NULL,
            tag_value TEXT NOT NULL,
            PRIMARY KEY (resource_id, tag_type, tag_value)
        );

        CREATE INDEX IF NOT EXISTS idx_resource_tags_type_value ON resource_tags(tag_type, tag_value);
        CREATE INDEX IF NOT EXISTS idx_resource_tags_resource ON resource_tags(resource_id);
        CREATE INDEX IF NOT EXISTS idx_resources_kind ON resources(resource_kind);
        CREATE INDEX IF NOT EXISTS idx_resources_active ON resources(is_active);
    "#),
