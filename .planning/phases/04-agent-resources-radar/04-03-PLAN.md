    ```typescript
    import { writable, derived } from "svelte/store";
    import type { ResourceCardDto, ResourceListRequest } from "$lib/types";
    import { listResources, searchResources, curateResource } from "$lib/ipc/tauri";

    // ── State ──────────────────────────────────────────────

    type ResourcesState = {
      items: ResourceCardDto[];
      loading: boolean;
      error: string | null;
      filters: ResourceListRequest;
      availableTags: string[];  // 所有可用标签（从 items 中提取）
    };

    const initialState: ResourcesState = {
      items: [],
      loading: false,
      error: null,
      filters: {},
      availableTags: [],
    };

    // ── Store ──────────────────────────────────────────────

    const _state = writable<ResourcesState>(initialState);

    export const resources = derived(_state, ($s) => $s.items);
    export const resourcesLoading = derived(_state, ($s) => $s.loading);
    export const resourcesError = derived(_state, ($s) => $s.error);
    export const resourceFilters = derived(_state, ($s) => $s.filters);
    export const availableResourceTags = derived(_state, ($s) => $s.availableTags);

    // ── Actions ──────────────────────────────────────────────

    /** 加载资源列表 */
    export async function loadResources() {
      _state.update((s) => ({ ...s, loading: true, error: null }));
      try {
        const items = await listResources(50, 0);
        const tags = extractAllTags(items);
        _state.update((s) => ({ ...s, items, loading: false, availableTags: tags }));
      } catch (e) {
        _state.update((s) => ({ ...s, loading: false, error: String(e) }));
      }
    }

    /** 按筛选条件搜索资源 */
    export async function filterResources(filters: ResourceListRequest) {
      _state.update((s) => ({ ...s, loading: true, error: null, filters }));
      try {
        const items = await searchResources(filters);
        _state.update((s) => ({ ...s, items, loading: false }));
      } catch (e) {
        _state.update((s) => ({ ...s, loading: false, error: String(e) }));
      }
    }

    /** 切换精选状态 */
    export async function toggleCurate(resourceId: string, isCurated: boolean) {
      try {
        const updated = await curateResource({
          resource_id: resourceId,
          action: isCurated ? "remove" : "add",
        });
        _state.update((s) => ({
          ...s,
          items: s.items.map((item) =>
            item.resource_id === resourceId ? updated : item
          ),
        }));
      } catch (e) {
        _state.update((s) => ({ ...s, error: String(e) }));
      }
    }

    /** 清除筛选 */
    export function clearFilters() {
      _state.update((s) => ({ ...s, filters: {} }));
      loadResources();
    }

    // ── Helpers ──────────────────────────────────────────────

    /** 从资源列表中提取所有唯一标签 */
    function extractAllTags(items: ResourceCardDto[]): string[] {
      const tagSet = new Set<string>();
      for (const item of items) {
        item.languages.forEach((l) => tagSet.add(l));
        item.framework_tags.forEach((t) => tagSet.add(t));
        item.agent_tags.forEach((t) => tagSet.add(t));
      }
      return Array.from(tagSet).sort();
    }
    ```

    2. 创建 `apps/desktop-ui/src/lib/components/ResourceFilters.svelte`:

    ```svelte
    <script lang="ts">
      import type { ResourceListRequest } from "$lib/types";

      export let availableTags: string[];
      export let currentFilters: ResourceListRequest;
      export let onFilter: (filters: ResourceListRequest) => void;
      export let onClear: () => void;

      const resourceKinds = [
        { value: "", label: "全部类型" },
        { value: "MCP_SERVER", label: "🔌 MCP Server" },
        { value: "SKILL", label: "⚡ Skill" },
        { value: "AGENT_FRAMEWORK", label: "🤖 Agent Framework" },
      ];

      let selectedKind = currentFilters.resource_kind || "";
      let selectedTag = currentFilters.tag_value || "";
      let selectedLanguage = currentFilters.language || "";

      function applyFilters() {
        const filters: ResourceListRequest = {};
        if (selectedKind) filters.resource_kind = selectedKind;
        if (selectedTag) {
          filters.tag_type = "free";
          filters.tag_value = selectedTag;
        }
        if (selectedLanguage) filters.language = selectedLanguage;
        onFilter(filters);
      }

      function handleClear() {
        selectedKind = "";
        selectedTag = "";
        selectedLanguage = "";
        onClear();
      }
    </script>

    <div class="flex flex-wrap items-center gap-2 p-3 rounded-[var(--radius)] bg-[var(--surface)]">
      <!-- 类型筛选 -->
      <select bind:value={selectedKind} on:change={applyFilters}
              class="text-sm bg-[var(--bg)] border border-[var(--border)] rounded px-2 py-1">
        {#each resourceKinds as kind}
          <option value={kind.value}>{kind.label}</option>
        {/each}
      </select>

      <!-- 语言筛选 -->
      <select bind:value={selectedLanguage} on:change={applyFilters}
              class="text-sm bg-[var(--bg)] border border-[var(--border)] rounded px-2 py-1">
        <option value="">全部语言</option>
        {#each availableTags.filter(t => !['mcp','skill','agent'].includes(t.toLowerCase())) as lang}
          <option value={lang}>{lang}</option>
        {/each}
      </select>

      <!-- 标签筛选 -->
      {#if availableTags.length > 0}
        <div class="flex flex-wrap gap-1">
          {#each availableTags.slice(0, 8) as tag}
            <button
              class="text-xs px-2 py-0.5 rounded border
                {selectedTag === tag ? 'bg-[var(--accent)] border-[var(--accent)]' : 'border-[var(--border)] hover:bg-[var(--surface)]'}"
              on:click={() => { selectedTag = selectedTag === tag ? "" : tag; applyFilters(); }}
            >
              {tag}
            </button>
          {/each}
        </div>
      {/if}

      <!-- 清除按钮 -->
      {#if selectedKind || selectedTag || selectedLanguage}
        <button class="text-xs text-muted hover:text-foreground" on:click={handleClear}>
          清除筛选
        </button>
      {/if}
    </div>
    ```
