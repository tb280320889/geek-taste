       ```rust
       use shared_contracts::error_dto::{AppErrorDto, ErrorKind};
       
       pub fn classify_error(e: &str) -> AppErrorDto {
           let lower = e.to_lowercase();
           if lower.contains("401") || lower.contains("403") || lower.contains("unauthorized") {
               AppErrorDto { code: ErrorKind::AuthExpired, message: e.to_string() }
           } else if lower.contains("404") || lower.contains("not found") {
               AppErrorDto { code: ErrorKind::NotFound, message: e.to_string() }
           } else if lower.contains("rate limit") || lower.contains("429") {
               AppErrorDto { code: ErrorKind::RateLimited, message: e.to_string() }
           } else if lower.contains("network") || lower.contains("dns") || lower.contains("connection")
               || lower.contains("timeout") || lower.contains("fetch") {
               AppErrorDto { code: ErrorKind::NetworkError, message: e.to_string() }
           } else {
               AppErrorDto { code: ErrorKind::Internal, message: e.to_string() }
           }
       }
       ```
    4. 创建 `crates/runtime_tauri/src/commands/sync_status.rs` — 返回各数据源最后同步时间:

       ```rust
       use serde::Serialize;
       use tauri::AppHandle;
       use super::helpers::get_db_connection;
       
       #[derive(Debug, Clone, Serialize)]
       pub struct SyncStatusDto {
           pub is_online: bool,
           pub last_topk_sync: Option<String>,
           pub last_signal_sync: Option<String>,
       }
       
       #[tauri::command]
       pub async fn get_sync_status(app: AppHandle) -> Result<SyncStatusDto, String> {
           let conn = get_db_connection(&app)?;
           let last_topk = persistence_sqlite::get_last_snapshot_time(&conn).ok();
           let last_signal = persistence_sqlite::get_last_signal_time(&conn).ok();
           // is_online 由前端检测后通过 IPC 更新，后端默认 true
           Ok(SyncStatusDto {
               is_online: true,
               last_topk_sync: last_topk,
               last_signal_sync: last_signal,
           })
       }
       ```
       注意: 如果 persistence_sqlite 中没有 `get_last_snapshot_time` / `get_last_signal_time` 函数，在 persistence 层添加简单的 `SELECT MAX(created_at) FROM ranking_snapshots` / `SELECT MAX(created_at) FROM signals` 查询。
    5. 在 `crates/runtime_tauri/src/commands/mod.rs` 添加 `pub mod sync_status;`

       ```typescript
       import { writable, derived } from "svelte/store";
       import { getSyncStatus } from "$lib/ipc/tauri";
       
       export const isOnline = writable(true);
       export const lastSyncedAt = writable<Record<string, string | null>>({});
       
       // STALE 阈值(毫秒): 排名 1h, 信号 30min
       const STALE_THRESHOLDS: Record<string, number> = {
         topk: 60 * 60 * 1000,
         signals: 30 * 60 * 1000,
       };
       
       export function isStale(source: string): boolean {
         // 由 derived store 在组件中使用 — 这里导出检查函数
         return false; // 实际逻辑在组件中通过 lastSyncedAt 计算
       }
       
       /** 从 IPC 错误判断网络状态 */
       export function handleIpcError(err: unknown): void {
         const msg = String(err);
         if (msg.includes("NETWORK_ERROR") || msg.includes("network") || msg.includes("fetch failed")) {
           isOnline.set(false);
         }
       }
       
       /** 应用启动时检测网络状态 */
       export async function checkNetworkStatus(): Promise<void> {
         try {
           const status = await getSyncStatus();
           isOnline.set(status.is_online);
           lastSyncedAt.set({
             topk: status.last_topk_sync,
             signals: status.last_signal_sync,
           });
         } catch {
           isOnline.set(false);
         }
       }
       ```
